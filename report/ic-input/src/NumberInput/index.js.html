/**
 * @name: ic-input ;
 * @author: sophiehui ;
 * @description: ic-input for ifchange ;
 * */
 
import React,{PureComponent,Fragment} from 'react'
import classnames from 'classnames'
import IcIcon from '@eui/ic-iconfont'
import Input from '../baseInput'
import {calculateWidth} from '../calculateWidth'
import './style.scss'
 
class NumberInput extends PureComponent{
    handleChange=(e,data)=>{
        const {onChange,min,max,value} = this.props
        let newValue = parseInt(data.replace(/[^0-9]/g,''))||'0'||''
        if(newValue>max){
            newValue = value
        }
        if(newValue<min){
            newValue = min
        }
        onChange(e,newValue)
    }
    handlerReduction = () =>{
        const {value,onChange,min} = this.props
        if(parseInt(value)>min){
            let newValue = parseInt(value)-1
            onChange(null,newValue)
        }
    }
    handlerAdd = () =>{
        const {value,onChange,max} = this.props
        if(parseInt(value)<max){
            let newValue = parseInt(value)+1
            onChange(null,newValue)
        }
    }
    render() {
        const {
            placeholder,
            className,
            width:propsWidth,
            style,
            onPressEnter,
            disabled,
            value,
            onChange,
            defaultValue,
            max,
            size,
            min,
            type,
            ...args} = this.props;
        const width = calculateWidth(propsWidth)
        return (
            <Fragment>
                <div className={classnames("eui-ic-number-input-wrapper",className)} style = {style}>
                    <span className={classnames("eui-ic-number-input",{"number-input_disable":disabled})} style = {{"width":width}}>
                        <IcIcon type = "video_hidden" className ={classnames("number-input-reduction",{"chang-disable":value<=min})} onClick={this.handlerReduction}></IcIcon>
                        <Input size="default" {...args} placeholder = {placeholder} onChange={this.handleChange} value={value} className="eui-ic-input-number__text"></Input>
                        <IcIcon type = "add2" className = {classnames("number-input-add",{"chang-disable":value>=max})} onClick={this.handlerAdd}></IcIcon>
                    </span>
                </div>
            </Fragment>
        );
    }
}
 
export default NumberInput;
