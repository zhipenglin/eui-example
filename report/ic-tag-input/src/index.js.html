/**
 * @name: ic-tag-input ;
 * @author: sophiehui ;
 * @description: ic-tag-input for ifchange ;
 * */
 
import React,{PureComponent,Fragment} from 'react'
import classnames from 'classnames'
import PropTypes from 'prop-types'
import './style.scss'
import IcTag from '@eui/ic-tag'
 
class icTagInput extends PureComponent{
    state={
        tags:[],
        currentValue:'',
        isFocus:false
    }
    static defaultProps={
        maxLength:'20',
        maxHeight:'200',
        placeholder:'',
        tagList:[],
        value:'',
        width:400,
        onChange:()=>{},
        onBlur:()=>{},
        onFocus:()=>{}
    };
    static propTypes={
        maxLength:PropTypes.oneOfType([
            PropTypes.string,
            PropTypes.number
        ]),
        width:PropTypes.oneOfType([
            PropTypes.string,
            PropTypes.number
        ]),
        className:PropTypes.string,
        placeholder:PropTypes.string,
        style:PropTypes.object,
        tagList:PropTypes.array,
        value:PropTypes.string,
        onChange:PropTypes.func,
    };
    constructor(props) {
        super(props);
        this.state = {
            tags: props.tagList,
            currentValue:props.value||''
        }
    }
    handChange = (e) =>{
        const {maxLength,onChange} = this.props;
        this.setState({
            currentValue:maxLength?e.target.value.replace(/(^\s*)|(\s*$)/g,"").slice(0,maxLength):e.target.value
        })
        onChange({
            value:this.state.currentValue,
            tagList:this.state.tags
        });
    }
    handlePress = (e)=>{
        let tagArray = this.state.tags.slice(0)
        if(e.keyCode == '8'&&!e.target.value){
            this.setState({
                tags:tagArray.slice(0,-1),
            })
        }
    }
    handleFocus=(e)=>{
        const {onFocus} = this.props
        this.setState({
            isFocus:true
        })
        onFocus(e)
    }
    handleBlur = (e)=>{
        const {onBlur} = this.props
        this.setState({
            isFocus:false
        })
        onBlur(e)
    }
    handleEnter = (e) =>{
        let value = this.state.currentValue;
        if (e.keyCode == '13'&&value) {
            let tagArray = this.state.tags.slice(0)
            tagArray.push(value)
            this.setState({
                tags:tagArray,
                currentValue:''
            })
        }
    }
    render() {
        const {tags,currentValue,isFocus} = this.state
        const {maxLength,maxHeight,tagList,alue,width,onChange,placeholder,className,style,...args} = this.props
        let re = /^[1-9]+[0-9]*]*$/
        //tag输入框的最大高度
        let newMaxHeight = '200px'
        Eif(maxHeight&&re.test(maxHeight)){
            newMaxHeight = maxHeight+'px'
        }
        //tag输入框的宽度
        let newWidth = "400px"
        Eif(width&&re.test(width)){
            newWidth = width+'px'
        }
        //tags为空，含有当前输入框里面有值
        let hasContent = !!(tags.length=='0'&&currentValue);
        let tagClass = classnames("eui-ic-tag-input",{"eui-ic-tag-fix-height":tags.length>0||currentValue},{"eui-ic-tag-focus":isFocus},{
            "eui-ic-tag-fix-position":hasContent},className)
        let newStyle = Object.assign({},{maxHeight:newMaxHeight,width:newWidth},style)
        return (
            <Fragment>
                <div className={tagClass} style={newStyle}>
                    {tags.length>0?tags.map((item,index)=><span
                        key = {index}
                        className={classnames("tag-wrapper")}
                    ><IcTag color="#3B99FF" closable={true}>{item}</IcTag></span>):''}
                    <span className="tag-input-wrapper"><input {...args} placeholder = {tags.length=='0'&&!currentValue?placeholder:''} onFocus={this.handleFocus} onBlur = {this.handleBlur} maxLength={maxLength} className="input-inner" onKeyUp={this.handleEnter} onChange = {this.handChange} onKeyDown = {this.handlePress} value = {currentValue}></input></span>
                </div>
            </Fragment>
        );
    }
}
 
export default icTagInput;
