/**
 * @name: ic-tooltip ;
 * @author: suzhen.han ;
 * @description: 气泡提示组件 ;
 * */
import React, {PureComponent} from 'react';
import {createPortal} from 'react-dom';
import classnames from 'classnames';
import './index.scss';
import placementFn from './placement';
 
const prefix = 'eui-tooltip';
 
class Pop extends PureComponent {
    constructor() {
        super(...arguments);
        this.state = {
            selfWidth: 0, // 为了计算气泡的位置，必须得到气泡的宽高数据，组件完成挂载后更改此数据
            selfHeight: 0
        }
        this.pop = React.createRef();
        const container = this.props.getTooltipContainer();
        const doc = window.document;
        this.node = doc.createElement('div');
        this.node.className = prefix + '-wrapper';
 
        if(container && container.nodeType === 1){
            container.appendChild(this.node);
        }else {
            throw new Error('getTooltipContainer的返回值必须为一个元素节点');
        }
    }
 
    componentDidMount() {
        const pop = this.pop.current;
        this.setState({
            selfWidth: pop.clientWidth,
            selfHeight: pop.clientHeight
        });
    }
 
    componentWillUnmount() {
        this.props.getTooltipContainer().removeChild(this.node);
    }
 
    render() {
        const {title, placement, wrapperClass, show, position, autoAdjustOverflow, arrowPointAtCenter, getTooltipContainer, isFirstRender, ...others} = this.props;
        const {selfWidth, selfHeight} = this.state; // 气泡自身宽高
        // 通过计算获取气泡位置信息
        const popInfo = placementFn({
            placement,
            targetInfo: position,
            selfInfo: {width: selfWidth, height: selfHeight},
            autoAdjustOverflow,
            arrowPointAtCenter
        });
 
        return createPortal(
            <div className={classnames(prefix, {
                [wrapperClass]: wrapperClass,
                [prefix + "-placement-" + popInfo.placement]: popInfo.placement,
                [prefix + "-hidden"]: !show
            })}
                 style={{left: popInfo.left, top: popInfo.top}}
                 ref={this.pop}
                 {...others}
            >
                <div className={classnames(prefix + "-arrow", {
                    [prefix + "-arrow-" + popInfo.placement]: popInfo.placement
                })}>
 
                </div>
                <div className={`${prefix}-inner`}>{title}</div>
            </div>,
            this.node
        );
    }
}
 
export default Pop;
