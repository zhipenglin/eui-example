import React from 'react';
function coordMap(x, y, R, a) {
 
    var ta = (360 - a) * Math.PI / 180,
        tx, ty;
    tx = R * Math.cos(ta)
    ty = R * Math.sin(ta)
    return {
        x: x + tx,
        y: y - ty
    }
}
 
function drawPath(data) {
    var path,
        s = new coordMap(data.x, data.y, data.R, data.startAngle),
        e = new coordMap(data.x, data.y, data.R, data.endAngle),
    // 画一段到(x,y)的椭圆弧. 椭圆弧的 x, y 轴半径分别为 rx,ry. 椭圆相对于 x 轴旋转 x-axis-rotation 度. large-arc=0表明弧线小于180读, large-arc=1表示弧线大于180度. sweep=0表明弧线逆时针旋转, sweep=1表明弧线顺时间旋转.
    // svg : [A | a] (rx ry x-axis-rotation large-arc-flag sweep-flag x y)+ 
    path = 'M ' + s.x + ' ' + s.y + ' A ' + data.R + ' ' + data.R + ' 0' + ' 0' + ' 1 ' + e.x + ' ' + e.y;
 
    return (
        <path 
            d={path}
            fill='none'
            stroke={data.color || '#CCC'}
            strokeWidth={data.strokeWidth || 1}
            strokeLinecap={data.strokelinecap ? data.strokelinecap: ''}
            color={data.color || '#FF8833'}
            strokeDasharray={data.strokeDasharray ? data.strokeDasharray: ''}
            transform={data.transform ? data.transform : ''}
        />)
}
function initSvg() {
    return generateSVG()
}
function generateSVG () {
    return (<svg
        xmlns='http://www.w3.org/2000/svg'
        version='1.1'
        width='30px'
        height='30px'
    >
        {drawPath({
            startAngle: 0,
            endAngle: 0,
            x: 15,
            y: 15,
            R: 10,
            strokelinecap: 'round',
            color: '#FFF',
            strokeWidth: 3,
            transform: 'rotate(-90, 15, 15)'
        })}
    </svg>)
}
function drawSvg(p1, p2, color, absPercent) {
    let endAngle = (360 * p1).toFixed(3)
    let startAngle = (360 * p2).toFixed(3)
    // endAngle = startAngle - endAngle
    // startAngle = 0
    const rotate = -90 + absPercent * 360
    return (<svg
        xmlns='http://www.w3.org/2000/svg'
        version='1.1'
        width='30px'
        height='30px'
    >
            {drawPath({
                startAngle: startAngle,
                endAngle: endAngle,
                x: 15,
                y: 15,
                R: 10,
                strokelinecap: 'round',
                strokeWidth: 3,
                color: color || '#FF8833',
                stroke: color || '#FF8833',
                transform: `rotate(${rotate}, 15, 15)`
        })}
    </svg>)
};
 
export default drawSvg
export {initSvg}
